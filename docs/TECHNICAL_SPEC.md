def attribute_orders_to_ads(orders, vk_stats, attribution_window=7):
    """
    –°–≤—è–∑–∞—Ç—å –∑–∞–∫–∞–∑—ã —Å —Ä–µ–∫–ª–∞–º–Ω—ã–º–∏ –∫–∞–º–ø–∞–Ω–∏—è–º–∏

        vk_stats: —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞ VK –∑–∞ –ø–µ—Ä–∏–æ–¥
        attribution_window: –æ–∫–Ω–æ –∞—Ç—Ä–∏–±—É—Ü–∏–∏ (–¥–Ω–µ–π)
    """
    attributed = []
    
    for order in orders:
        order_date = order.creation_date
        window_start = order_date - timedelta(days=attribution_window)
        
        # –ù–∞–π—Ç–∏ —Ä–µ–∫–ª–∞–º—É –≤ –æ–∫–Ω–µ –∞—Ç—Ä–∏–±—É—Ü–∏–∏

            date__gte=window_start,
            date__lte=order_date
        )
        
        if relevant_ads.exists():
            # –†–∞—Å–ø—Ä–µ–¥–µ–ª–∏—Ç—å –∑–∞–∫–∞–∑ –ø—Ä–æ–ø–æ—Ä—Ü–∏–æ–Ω–∞–ª—å–Ω–æ –∫–ª–∏–∫–∞–º
            total_clicks = sum(ad.clicks for ad in relevant_ads)
            
            for ad in relevant_ads:

                    'order_id': order.id,
                    'campaign_id': ad.campaign_id,
                    'revenue_share': order.total * weight,

def auto_map_products(c1_products, yandex_products, threshold=0.8):
    """

    """

    for c1_prod in c1_products:
        best_match = None
        best_score = 0
        
        for yandex_prod in yandex_products:
            # –ü–æ–ø—ã—Ç–∫–∞ 1: —Ç–æ—á–Ω–æ–µ —Å–æ–≤–ø–∞–¥–µ–Ω–∏–µ –∞—Ä—Ç–∏–∫—É–ª–∞
            if c1_prod.article and c1_prod.article == yandex_prod.offer_id:
                mappings.append({
                    'c1_code': c1_prod.code,
                    'yandex_offer_id': yandex_prod.offer_id,
                    'confidence': 1.0,
                    'method': 'exact_article'
                })
                break
            
            # –ü–æ–ø—ã—Ç–∫–∞ 2: —Å–æ–≤–ø–∞–¥–µ–Ω–∏–µ –Ω–∞–∑–≤–∞–Ω–∏—è
            similarity = SequenceMatcher(
                None,
                c1_prod.name.lower(),
                yandex_prod.name.lower()

            if similarity > best_score:
                best_score = similarity
                best_match = yandex_prod
        
        # –ï—Å–ª–∏ –Ω–∞—à–ª–∏ —Ö–æ—Ä–æ—à–µ–µ —Å–æ–≤–ø–∞–¥–µ–Ω–∏–µ –ø–æ –Ω–∞–∑–≤–∞–Ω–∏—é
        if best_match and best_score >= threshold:
            mappings.append({
                'c1_code': c1_prod.code,
                'yandex_offer_id': best_match.offer_id,
                'confidence': round(best_score, 2),
                'method': 'name_similarity'
            })
